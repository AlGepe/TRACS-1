228  {
229  unwrapped_stepper_type &stepper = this->m_stepper;
230  const size_t max_attempts = 1000;
231  size_t trials = 0;
232  controlled_step_result res = success;

235  res = stepper.try_step( this->m_system , *( this->m_state ) , this->m_t , dt );
