151  {
152  BOOST_USING_STD_MIN();
153  BOOST_USING_STD_MAX();
154  using std::pow;
155 
156  static const value_type val1( 1.0 );
157 
158  bool reject( true );
159 
160  time_vector h_opt( m_k_max+1 );
161  inv_time_vector work( m_k_max+1 );
162 
163  m_k_final = 0;
164  time_type new_h = dt;
165 
166  //std::cout << "t=" << t <<", dt=" << dt << ", k_opt=" << m_current_k_opt << ", first: " << m_first << std::endl;
167 
168  for( size_t k = 0 ; k <= m_current_k_opt+1 ; k++ )
169  {
170  m_midpoint.set_steps( m_interval_sequence[k] );
171  if( k == 0 )
172  {
173  m_midpoint.do_step( system , in , dxdt , t , out , dt , m_mp_states[k].m_v , m_derivs[k]);
174  }
175  else
176  {
177  m_midpoint.do_step( system , in , dxdt , t , m_table[k-1].m_v , dt , m_mp_states[k].m_v , m_derivs[k] );
178  extrapolate( k , m_table , m_coeff , out );
179  // get error estimate
180  m_algebra.for_each3( m_err.m_v , out , m_table[0].m_v ,
181  typename operations_type::template scale_sum2< value_type , value_type >( val1 , -val1 ) );
182  const value_type error = m_error_checker.error( m_algebra , in , dxdt , m_err.m_v , dt );
183  h_opt[k] = calc_h_opt( dt , error , k );
184  work[k] = static_cast<value_type>( m_cost[k] ) / h_opt[k];
185 
186  m_k_final = k;
187 
188  if( (k == m_current_k_opt-1) || m_first )
189  { // convergence before k_opt ?
190  if( error < 1.0 )
191  {
192  //convergence
193  reject = false;
194  if( (work[k] < KFAC2*work[k-1]) || (m_current_k_opt <= 2) )
195  {
196  // leave order as is (except we were in first round)
197  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k)+1 ) );

199  } else {
200  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(k) ) );
201  new_h = h_opt[k];
202  }
203  break;
204  }
205  else if( should_reject( error , k ) && !m_first )
206  {
207  reject = true;
208  new_h = h_opt[k];
209  break;
210  }
211  }
212  if( k == m_current_k_opt )
213  { // convergence at k_opt ?
214  if( error < 1.0 )
215  {
216  //convergence
217  reject = false;
218  if( (work[k-1] < KFAC2*work[k]) )
219  {
220  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
221  new_h = h_opt[m_current_k_opt];
222  }
223  else if( (work[k] < KFAC2*work[k-1]) && !m_last_step_rejected )
224  {
225  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , static_cast<int>(m_current_k_opt)+1 );
226  new_h = h_opt[k]*static_cast<value_type>( m_cost[m_current_k_opt] ) / static_cast<value_type>( m_cost[k] );
227  } else
228  new_h = h_opt[m_current_k_opt];
229  break;
230  }
231  else if( should_reject( error , k ) )
232  {
233  reject = true;
234  new_h = h_opt[m_current_k_opt];
235  break;
236  }
237  }
238  if( k == m_current_k_opt+1 )
239  { // convergence at k_opt+1 ?
240  if( error < 1.0 )
241  { //convergence
242  reject = false;
243  if( work[k-2] < KFAC2*work[k-1] )
244  m_current_k_opt = max BOOST_PREVENT_MACRO_SUBSTITUTION( 2 , static_cast<int>(m_current_k_opt)-1 );
245  if( (work[k] < KFAC2*work[m_current_k_opt]) && !m_last_step_rejected )
246  m_current_k_opt = min BOOST_PREVENT_MACRO_SUBSTITUTION( static_cast<int>(m_k_max)-1 , static_cast<int>(k) );
247  new_h = h_opt[m_current_k_opt];
248  } else
249  {
250  reject = true;
251  new_h = h_opt[m_current_k_opt];
252  }
253  break;
254  }
255  }
256  }
257 
258  if( !reject )
259  {
260 
261  //calculate dxdt for next step and dense output
262  typename odeint::unwrap_reference< System >::type &sys = system;
263  sys( out , dxdt_new , t+dt );
264 
265  //prepare dense output
266  value_type error = prepare_dense_output( m_k_final , in , dxdt , out , dxdt_new , dt );
267 
268  if( error > static_cast<value_type>(10) ) // we are not as accurate for interpolation as for the steps
269  {
270  reject = true;
271  new_h = dt * pow BOOST_PREVENT_MACRO_SUBSTITUTION( error , static_cast<value_type>(-1)/(2*m_k_final+2) );
272  } else {
273  t += dt;
274  }
275  }
276  //set next stepsize
277  if( !m_last_step_rejected || (new_h < dt) )
278  dt = new_h;
279 
280  m_last_step_rejected = reject;
281  if( reject )
282  return fail;
283  else
284  return success;
285  }

302  std::pair< time_type , time_type > do_step( System system )
303  {
304  const size_t max_count = 1000;
305 
306  if( m_first )
307  {
308  typename odeint::unwrap_reference< System >::type &sys = system;
309  sys( get_current_state() , get_current_deriv() , m_t );
310  }
311 
312  controlled_step_result res = fail;
313  m_t_last = m_t;
314  size_t count = 0;
315  while( res == fail )
316  {
317  res = try_step( system , get_current_state() , get_current_deriv() , m_t , get_old_state() , get_old_deriv() , m_dt );
318  m_first = false;
319  if( count++ == max_count )
320  throw std::overflow_error( "bulirsch_stoer : too much iterations!");
321  }
322  toggle_current_state();
323  return std::make_pair( m_t_last , m_t );

409  time_type calc_h_opt( time_type h , value_type error , size_t k ) const
410  {
411  BOOST_USING_STD_MIN();
412  BOOST_USING_STD_MAX();
413  using std::pow;
414 
415  value_type expo=1.0/(m_interval_sequence[k-1]);
416  value_type facmin = pow BOOST_PREVENT_MACRO_SUBSTITUTION( STEPFAC3 , expo );
417  value_type fac;
418  if (error == 0.0)
419  fac=1.0/facmin;
420  else
421  {
422  fac = STEPFAC2 / pow BOOST_PREVENT_MACRO_SUBSTITUTION( error / STEPFAC1 , expo );
423  fac = max BOOST_PREVENT_MACRO_SUBSTITUTION( facmin/STEPFAC4 , min BOOST_PREVENT_MACRO_SUBSTITUTION( 1.0/facmin , fac ) );
424  }
425  return h*fac;
426  }

435  bool should_reject( value_type error , size_t k ) const
436  {
437  if( k == m_current_k_opt-1 )
438  {
439  const value_type d = m_interval_sequence[m_current_k_opt] * m_interval_sequence[m_current_k_opt+1] /
440  (m_interval_sequence[0]*m_interval_sequence[0]);
441  //step will fail, criterion 17.3.17 in NR
442  return ( error > d*d );
443  }
444  else if( k == m_current_k_opt )
445  {
446  const value_type d = m_interval_sequence[m_current_k_opt+1] / m_interval_sequence[0];
447  return ( error > d*d );
448  } else
449  return error > 1.0;

452  template< class StateIn1 , class DerivIn1 , class StateIn2 , class DerivIn2 >
453  value_type prepare_dense_output( int k , const StateIn1 &x_start , const DerivIn1 &dxdt_start ,
454  const StateIn2 & /* x_end */ , const DerivIn2 & /*dxdt_end */ , time_type dt )
455  /* k is the order to which the result was approximated */
456  {
457 
458  /* compute the coefficients of the interpolation polynomial
459  * we parametrize the interval t .. t+dt by theta = -1 .. 1
460  * we use 2k+3 values at the interval center theta=0 to obtain the interpolation coefficients
461  * the values are x(t+dt/2) and the derivatives dx/dt , ... d^(2k+2) x / dt^(2k+2) at the midpoints
462  * the derivatives are approximated via finite differences
463  * all values are obtained from interpolation of the results from the increasing orders of the midpoint calls
464  */
465 
466  // calculate finite difference approximations to derivatives at the midpoint
467  for( int j = 0 ; j<=k ; j++ )
468  {
469  /* not working with boost units... */
470  const value_type d = m_interval_sequence[j] / ( static_cast<value_type>(2) * dt );
471  value_type f = 1.0; //factor 1/2 here because our interpolation interval has length 2 !!!
472  for( int kappa = 0 ; kappa <= 2*j+1 ; ++kappa )
473  {
474  calculate_finite_difference( j , kappa , f , dxdt_start );
475  f *= d;
476  }
477 
478  if( j > 0 )
479  extrapolate_dense_out( j , m_mp_states , m_coeff );
480  }
481 
482  time_type d = dt/2;
483 
484  // extrapolate finite differences
485  for( int kappa = 0 ; kappa<=2*k+1 ; kappa++ )
486  {
487  for( int j=1 ; j<=(k-kappa/2) ; ++j )
488  extrapolate_dense_out( j , m_diffs[kappa] , m_coeff , kappa/2 );
489 
490  // extrapolation results are now stored in m_diffs[kappa][0]
491 
492  // divide kappa-th derivative by kappa because we need these terms for dense output interpolation
493  m_algebra.for_each1( m_diffs[kappa][0].m_v , typename operations_type::template scale< time_type >( static_cast<time_type>(d) ) );
494 
495  d *= dt/(2*(kappa+2));
496  }
497 
498  // dense output coefficients a_0 is stored in m_mp_states[0], a_i for i = 1...2k are stored in m_diffs[i-1][0]
499 
500  // the error is just the highest order coefficient of the interpolation polynomial
501  // this is because we use only the midpoint theta=0 as support for the interpolation (remember that theta = -1 .. 1)
502 
503  value_type error = 0.0;
504  if( m_control_interpolation )
505  {
506  boost::numeric::odeint::copy( m_diffs[2*k+1][0].m_v , m_err.m_v );
507  error = m_error_checker.error( m_algebra , x_start , dxdt_start , m_err.m_v , dt );
508  }
509 
510  return error;
511  }

513  template< class DerivIn >
514  void calculate_finite_difference( size_t j , size_t kappa , value_type fac , const DerivIn &dxdt )
515  {
516  const int m = m_interval_sequence[j]/2-1;
517  if( kappa == 0) // no calculation required for 0th derivative of f
518  {
519  m_algebra.for_each2( m_diffs[0][j].m_v , m_derivs[j][m].m_v ,
520  typename operations_type::template scale_sum1< value_type >( fac ) );
521  }
522  else
523  {
524  // calculate the index of m_diffs for this kappa-j-combination
525  const int j_diffs = j - kappa/2;
526 
527  m_algebra.for_each2( m_diffs[kappa][j_diffs].m_v , m_derivs[j][m+kappa].m_v ,
528  typename operations_type::template scale_sum1< value_type >( fac ) );
529  value_type sign = -1.0;
530  int c = 1;
531  //computes the j-th order finite difference for the kappa-th derivative of f at t+dt/2 using function evaluations stored in m_derivs
532  for( int i = m+static_cast<int>(kappa)-2 ; i >= m-static_cast<int>(kappa) ; i -= 2 )
533  {
534  if( i >= 0 )
535  {
536  m_algebra.for_each3( m_diffs[kappa][j_diffs].m_v , m_diffs[kappa][j_diffs].m_v , m_derivs[j][i].m_v ,
537  typename operations_type::template scale_sum2< value_type , value_type >( 1.0 ,
538  sign * fac * boost::math::binomial_coefficient< value_type >( kappa , c ) ) );
539  }
540  else
541  {
542  m_algebra.for_each3( m_diffs[kappa][j_diffs].m_v , m_diffs[kappa][j_diffs].m_v , dxdt ,

551  template< class StateOut >
552  void do_interpolation( time_type t , StateOut &out ) const
553  {
554  // interpolation polynomial is defined for theta = -1 ... 1
555  // m_k_final is the number of order-iterations done for the last step - it governs the order of the interpolation polynomial
556  const value_type theta = 2 * get_unit_value( (t - m_t_last) / (m_t - m_t_last) ) - 1;
557  // we use only values at interval center, that is theta=0, for interpolation
558  // our interpolation polynomial is thus of order 2k+2, hence we have 2k+3 terms
559 
560  boost::numeric::odeint::copy( m_mp_states[0].m_v , out );
561  // add remaining terms: x += a_1 theta + a2 theta^2 + ... + a_{2k} theta^{2k}
562  value_type theta_pow( theta );
563  for( size_t i=0 ; i<=2*m_k_final+1 ; ++i )
564  {
565  m_algebra.for_each3( out , out , m_diffs[i][0].m_v ,
566  typename operations_type::template scale_sum2< value_type >( static_cast<value_type>(1) , theta_pow ) );
567  theta_pow *= theta;
568  }
569  }
