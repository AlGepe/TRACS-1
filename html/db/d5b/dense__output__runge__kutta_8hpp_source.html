312  template< class System >
313  std::pair< time_type , time_type > do_step( System system )
314  {
315  const size_t max_count = 1000;
316 
317  if( !m_is_deriv_initialized )
318  {
319  typename odeint::unwrap_reference< System >::type &sys = system;
320  sys( get_current_state() , get_current_deriv() , m_t );
321  m_is_deriv_initialized = true;
322  }
323 
324  controlled_step_result res = fail;
325  m_t_old = m_t;
326  size_t count = 0;
327  do
328  {
329  res = m_stepper.try_step( system , get_current_state() , get_current_deriv() , m_t ,
330  get_old_state() , get_old_deriv() , m_dt );
331  if( count++ == max_count )
332  BOOST_THROW_EXCEPTION( std::overflow_error( "dense_output_controlled_explicit : too much iterations!") );
333  }
334  while( res == fail );
335  toggle_current_state();
336  return std::make_pair( m_t_old , m_t );
337  }
